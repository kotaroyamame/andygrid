Sample(option)使用例(オプション)

margin:0
cell:400
captionBg:rgba(255,255,255,0.5)
furufuru:true
rireki:true

    
    <script>
        $(document).on("ready",function(){
            $(".wrapSerector").andygrid({
                "margin":0,
                "cell":400,
                "captionBg":"rgba(255,255,255,0.5)",
                "furufuru":true,
                "rireki":true
            });
        });
    </script>

なお、履歴をリセットするボタンを設置したい場合は、下記のようにすることで設置できます。

    
    <script>
        $(document).on("ready",function(){
            $(".wrap--MarginZero").andygrid({
                "margin":0,
                "cell":400,
                "captionBg":"rgba(255,255,255,0.5)",
                "furufuru":true,
                "rireki":true
            });
            $("#リセットボタンのID").cokkieReset();
        });
    </script>
